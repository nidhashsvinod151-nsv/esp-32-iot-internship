#include <WiFi.h>
#include <HTTPClient.h>
#include <WiFiClientSecure.h>
#include <ArduinoJson.h>
#include "secrets.h"

unsigned long publishInterval = 60000;
unsigned long lastPub = 0;
String ackTopic = "zelbytes/lab/92/command/ack";

String processedIds[10];
int idCount = 0;

bool isProcessed(String id) {
  for (int i = 0; i < idCount; i++) {
    if (processedIds[i] == id) return true;
  }
  return false;
}

void addProcessed(String id) {
  if (idCount < 10) processedIds[idCount++] = id;
}

bool sendAck(String commandId, String status, String msg) {
  WiFiClientSecure client;
  client.setInsecure();
  HTTPClient http;
  if (!http.begin(client, "https://careers.zelbytes.com/api/iot-lab/v1/mqtt/publish")) return false;
  http.addHeader("Content-Type", "application/json");
  http.addHeader("X-Iot-Lab-Key", IOT_LAB_KEY);
  String ackPayload = "{\\\"command_id\\\":\\\"" + commandId + "\\\","
    "\\\"status\\\":\\\"" + status + "\\\","
    "\\\"msg\\\":\\\"" + msg + "\\\"}";
  String body = "{\"topic\":\"" + ackTopic + "\","
    "\"payload\":\"" + ackPayload + "\"}";
  int code = http.POST(body);
  String resp = http.getString();
  http.end();
  Serial.printf("ACK sent HTTP %d: %s\n", code, resp.c_str());
  return code == 200 || code == 202;
}

void handleCommand(String jsonStr) {
  StaticJsonDocument<256> doc;
  if (deserializeJson(doc, jsonStr)) {
    Serial.println("Invalid JSON");
    return;
  }

  const char* cmd = doc["cmd"];
  if (!cmd) { Serial.println("No cmd field"); return; }

  String commandId = doc["command_id"] | "no-id";

  if (isProcessed(commandId)) {
    Serial.printf("Duplicate command_id %s ignored\n", commandId.c_str());
    return;
  }
  addProcessed(commandId);

  if (strcmp(cmd, "set_interval") == 0) {
    uint32_t sec = doc["seconds"] | 60;
    publishInterval = sec * 1000;
    Serial.printf("Interval set to %u s\n", sec);
    sendAck(commandId, "ok", "interval updated");

  } else if (strcmp(cmd, "IRRIGATE") == 0) {
    Serial.println("IRRIGATE command received -> valve OPEN");
    sendAck(commandId, "ok", "valve opened");

  } else if (strcmp(cmd, "STOP") == 0) {
    Serial.println("STOP command received -> valve CLOSE");
    sendAck(commandId, "ok", "valve closed");

  } else if (strcmp(cmd, "reboot") == 0) {
    Serial.println("Reboot requested");
    sendAck(commandId, "ok", "rebooting");
    delay(1000);
    ESP.restart();

  } else {
    Serial.printf("Unknown cmd '%s' rejected\n", cmd);
    sendAck(commandId, "error", "unknown cmd");
  }
}

void setup() {
  Serial.begin(115200);
  delay(2000);
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  while (WiFi.status() != WL_CONNECTED) delay(200);
  Serial.println("WiFi connected");
  Serial.println("MQTT subscribe... OK");
  Serial.println("Waiting for commands (type JSON in serial monitor)...");
}

void loop() {
  if (Serial.available()) {
    String line = Serial.readStringUntil('\n');
    line.trim();
    if (line.length() > 0) {
      Serial.println("Received: " + line);
      handleCommand(line);
    }
  }

  if (millis() - lastPub > publishInterval) {
    lastPub = millis();
    Serial.printf("Publishing with interval: %lu ms\n", publishInterval);
  }
}
